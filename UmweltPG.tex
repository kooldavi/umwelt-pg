\begin{frame}{Graphbasierte Datenbank}
  \begin{itemize}
  	\item Speichert Daten in Knoten und Kanten
  	\item Kante verbindet immer zwei Knoten

  	\item Kanten stellen die Beziehung zwischen den Knoten dar
  \end{itemize}
\end{frame}
